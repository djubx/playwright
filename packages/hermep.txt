import 'package:flutter/material.dart';
import 'package:hermep/hermep.dart';

// Define a new Hermep app
class RealFlutter extends HermepApp {
  // Override the build method to define the app's UI
  @override
  Widget build(BuildContext context) {
    // Create a new Material app
    return MaterialApp(
      // Set the app's title
      title: 'Real Flutter',
      // Set the app's theme
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      // Set the app's home page
      home: MyHomePage(),
    );
  }
}

// Define a new home page
class MyHomePage extends StatefulWidget {
  // Create a new state object
  @override
  _MyHomePageState createState() => _MyHomePageState();
}

// Define a new state class
class _MyHomePageState extends State<MyHomePage> {
  // Initialize a counter variable
  int _counter = 0;

  // Define a method to increment the counter
  void _incrementCounter() {
    // Update the state object
    setState(() {
      _counter++;
    });
  }

  // Override the build method to define the page's UI
  @override
  Widget build(BuildContext context) {
    // Create a new Scaffold
    return Scaffold(
      // Set the page's app bar
      appBar: AppBar(
        title: Text('Real Flutter'),
      ),
      // Set the page's body
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: <Widget>[
            Text(
              'You have pushed the button this many times:',
            ),
            Text(
              '$_counter',
              style: Theme.of(context).textTheme.display1,
            ),
          ],
        ),
      ),
      // Set the page's floating action button
      floatingActionButton: FloatingActionButton(
        onPressed: _incrementCounter,
        tooltip: 'Increment',
        child: Icon(Icons.add),
      ),
    );
  }
}

// Define a main function to run the app
void main() {
  // Create a new Hermep app
  runApp(RealFlutter());
}

// The following comments explain the flow of the application:

// 1. The user launches the app, which creates a new instance of the RealFlutter class.
// 2. The RealFlutter class overrides the build method to define the app's UI, which consists of a Material app with a title, theme, and home page.
// 3. The home page is an instance of the MyHomePage class, which is a stateful widget that maintains a counter variable.
// 4. When the user presses the floating action button, the _incrementCounter method is called, which updates the state object and increments the counter variable.
// 5. The build method is called again to update the page's UI, which displays the new value of the counter variable.

// This is the end of the main file example.
