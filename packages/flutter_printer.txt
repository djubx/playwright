import 'package:flutter/material.dart';
import 'package:flutter_printer/flutter_printer.dart';

class RealFlutter extends StatefulWidget {
  @override
  _RealFlutterState createState() => _RealFlutterState();
}

class _RealFlutterState extends State<RealFlutter> {
  FlutterPrinter _printer;

  @override
  void initState() {
    super.initState();
    _printer = FlutterPrinter();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Flutter Printer Example'),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: <Widget>[
            ElevatedButton(
              onPressed: () {
                _printer.print('Hello, World!');
              },
              child: Text('Print Hello World'),
            ),
            SizedBox(height: 20),
            ElevatedButton(
              onPressed: () {
                _printer.printSettings(
                  paperSize: PaperSize.A4,
                  orientation: Orientation.Portrait,
                );
              },
              child: Text('Print Settings'),
            ),
          ],
        ),
      ),
    );
  }
}

// Application flow:
// 1. The user opens the app and is presented with a screen containing two buttons.
// 2. The user clicks the "Print Hello World" button, which calls the `_printer.print` method.
// 3. The `_printer.print` method prints the string "Hello, World!" to the printer.
// 4. The user clicks the "Print Settings" button, which calls the `_printer.printSettings` method.
// 5. The `_printer.printSettings` method sets the paper size to A4 and the orientation to portrait.
// 6. The user can then print their content using the `_printer.print` method.

// Final summary:
// The `flutter_printer` package provides a simple and easy-to-use API for printing documents, images, and other content from a Flutter app.
// The package supports multiple printing formats, including PDF and JPEG, and allows for customization of print settings, such as paper size and orientation.
// The example app demonstrates how to use the `flutter_printer` package to print a string and set print settings.
