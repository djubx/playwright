import 'package:flutter/material.dart';
import 'package:brazilian_banks/brazilian_banks.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Brazilian Banks Demo',
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: MyHomePage(),
    );
  }
}

class MyHomePage extends StatefulWidget {
  @override
  _MyHomePageState createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  final _formKey = GlobalKey<FormState>();
  final _bankCodeController = TextEditingController();
  final _bankNameController = TextEditingController();
  final _accountNumberController = TextEditingController();
  final _agencyNumberController = TextEditingController();

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Brazilian Banks Demo'),
      ),
      body: Padding(
        padding: const EdgeInsets.all(16.0),
        child: Form(
          key: _formKey,
          child: Column(
            children: [
              TextFormField(
                controller: _bankCodeController,
                decoration: InputDecoration(
                  labelText: 'Bank Code',
                ),
                validator: (value) {
                  if (value.isEmpty) {
                    return 'Please enter a bank code';
                  }
                  return null;
                },
              ),
              TextFormField(
                controller: _bankNameController,
                decoration: InputDecoration(
                  labelText: 'Bank Name',
                ),
                validator: (value) {
                  if (value.isEmpty) {
                    return 'Please enter a bank name';
                  }
                  return null;
                },
              ),
              TextFormField(
                controller: _accountNumberController,
                decoration: InputDecoration(
                  labelText: 'Account Number',
                ),
                validator: (value) {
                  if (value.isEmpty) {
                    return 'Please enter an account number';
                  }
                  return null;
                },
              ),
              TextFormField(
                controller: _agencyNumberController,
                decoration: InputDecoration(
                  labelText: 'Agency Number',
                ),
                validator: (value) {
                  if (value.isEmpty) {
                    return 'Please enter an agency number';
                  }
                  return null;
                },
              ),
              SizedBox(height: 16),
              ElevatedButton(
                onPressed: () {
                  if (_formKey.currentState.validate()) {
                    // Create a new instance of the RealFlutter class
                    RealFlutter realFlutter = RealFlutter();

                    // Get the bank name from the bank code
                    String bankName = realFlutter.getBankNameFromCode(_bankCodeController.text);
                    print(bankName);

                    // Get the bank code from the bank name
                    String bankCode = realFlutter.getBankCodeFromName(_bankNameController.text);
                    print(bankCode);

                    // Validate a bank account number
                    bool isValidAccount = realFlutter.validateAccountNumber(_accountNumberController.text, _bankCodeController.text);
                    print(isValidAccount);

                    // Validate a bank agency number
                    bool isValidAgency = realFlutter.validateAgencyNumber(_agencyNumberController.text, _bankCodeController.text);
                    print(isValidAgency);

                    // Generate a TED code
                    String tedCode = realFlutter.generateTedCode(_accountNumberController.text, _bankCodeController.text, _agencyNumberController.text);
                    print(tedCode);
                  }
                },
                child: Text('Submit'),
              ),
            ],
          ),
        ),
      ),
    );
  }
}

// The application flow is as follows:
// 1. The user enters a bank code, bank name, account number, and agency number in the form.
// 2. When the user submits the form, the application creates a new instance of the RealFlutter class.
// 3. The application uses the RealFlutter class to get the bank name from the bank code and vice versa.
// 4. The application validates the bank account number and agency number using the RealFlutter class.
// 5. The application generates a TED code using the RealFlutter class.
// 6. The application prints the results to the console.
