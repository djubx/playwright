```dart
import 'package:flutter/material.dart';
import 'package:vcard_maintained/vcard_maintained.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'vCard Demo',
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: MyHomePage(),
    );
  }
}

class MyHomePage extends StatefulWidget {
  @override
  _MyHomePageState createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  VCard _vcard;

  @override
  void initState() {
    super.initState();
    _createVCard();
  }

  void _createVCard() {
    // Create a new vCard
    VCard vcard = VCard(
      version: VCardVersion.v4,
      fn: 'John Doe',
      email: 'john.doe@example.com',
      tel: '+1234567890',
    );

    setState(() {
      _vcard = vcard;
    });
  }

  void _parseVCard() {
    // Parse a vCard from a string
    String vcardString = 'BEGIN:VCARD\r\nVERSION:4.0\r\nFN:John Doe\r\nEMAIL;TYPE=INTERNET:john.doe@example.com\r\nTEL;TYPE=VOICE:+1234567890\r\nEND:VCARD';
    VCard vcard = VCard.parse(vcardString);

    setState(() {
      _vcard = vcard;
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('vCard Demo'),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: <Widget>[
            Text(
              'vCard:',
              style: TextStyle(fontSize: 24),
            ),
            Text(
              _vcard != null ? _vcard.toString() : '',
              style: TextStyle(fontSize: 18),
            ),
            SizedBox(height: 20),
            ElevatedButton(
              onPressed: _createVCard,
              child: Text('Create vCard'),
            ),
            SizedBox(height: 10),
            ElevatedButton(
              onPressed: _parseVCard,
              child: Text('Parse vCard'),
            ),
          ],
        ),
      ),
    );
  }
}

// The application flow is as follows:
// 1. The user runs the app and is presented with a screen that displays a vCard.
// 2. The user can create a new vCard by pressing the "Create vCard" button.
// 3. The user can parse a vCard from a string by pressing the "Parse vCard" button.
// 4. The parsed vCard is displayed on the screen.
// 5. The user can repeat steps 2-4 as needed.

// The code flow is as follows:
// 1. The `MyApp` widget is created and run.
// 2. The `MyHomePage` widget is created and displayed.
// 3. The `_createVCard` method is called to create a new vCard.
// 4. The `_parseVCard` method is called to parse a vCard from a string.
// 5. The parsed vCard is displayed on the screen using the `Text` widget.
// 6. The user can interact with the app by pressing the buttons and viewing the vCard.
```
