import 'package:flutter/material.dart';
import 'package:viva_wallet_pos/viva_wallet_pos.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Viva Wallet POS Demo',
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: RealFlutter(),
    );
  }
}

class RealFlutter extends StatefulWidget {
  @override
  _RealFlutterState createState() => _RealFlutterState();
}

class _RealFlutterState extends State<RealFlutter> {
  VivaWalletPos _vivaWalletPos;
  PaymentRequest _paymentRequest;

  @override
  void initState() {
    super.initState();
    _vivaWalletPos = VivaWalletPos(
      clientId: 'YOUR_CLIENT_ID',
      clientSecret: 'YOUR_CLIENT_SECRET',
      posId: 'YOUR_POS_ID',
    );
    _paymentRequest = PaymentRequest(
      amount: 10.99,
      currency: 'EUR',
      paymentMethod: PaymentMethod.card,
    );
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Viva Wallet POS Demo'),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: <Widget>[
            Text(
              'Amount: ${_paymentRequest.amount} ${_paymentRequest.currency}',
            ),
            SizedBox(height: 20),
            ElevatedButton(
              onPressed: () {
                _vivaWalletPos.processPaymentRequest(_paymentRequest).then((paymentResponse) {
                  if (paymentResponse.status == PaymentStatus.success) {
                    // Payment successful
                    print('Payment successful');
                  } else {
                    // Payment failed
                    print('Payment failed');
                  }
                }).catchError((error) {
                  // Handle error
                  print('Error: $error');
                });
              },
              child: Text('Make Payment'),
            ),
          ],
        ),
      ),
    );
  }
}

// The application flow is as follows:
// 1. The user opens the app and is presented with a screen displaying the payment amount and a "Make Payment" button.
// 2. When the user clicks the "Make Payment" button, the app creates a payment request with the specified amount and currency.
// 3. The app then processes the payment request using the Viva Wallet POS instance.
// 4. If the payment is successful, the app displays a success message.
// 5. If the payment fails, the app displays an error message.
// 6. If an error occurs during the payment process, the app catches the error and displays an error message.
