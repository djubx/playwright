```dart
import 'package:flutter/material.dart';
import 'package:xor_encryption/xor_encryption.dart';

void main() {
  runApp(RealFlutter());
}

class RealFlutter extends StatefulWidget {
  @override
  _RealFlutterState createState() => _RealFlutterState();
}

class _RealFlutterState extends State<RealFlutter> {
  final _keyController = TextEditingController();
  final _dataController = TextEditingController();
  final _encryptedDataController = TextEditingController();
  final _decryptedDataController = TextEditingController();

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(
          title: Text('XOR Encryption Demo'),
        ),
        body: Padding(
          padding: const EdgeInsets.all(16.0),
          child: Column(
            children: [
              TextField(
                controller: _keyController,
                decoration: InputDecoration(
                  labelText: 'Enter Secret Key',
                  border: OutlineInputBorder(),
                ),
              ),
              SizedBox(height: 16),
              TextField(
                controller: _dataController,
                decoration: InputDecoration(
                  labelText: 'Enter Data to Encrypt',
                  border: OutlineInputBorder(),
                ),
              ),
              SizedBox(height: 16),
              ElevatedButton(
                onPressed: () {
                  final encryption = XorEncryption(key: _keyController.text);
                  final encryptedData = encryption.encrypt(_dataController.text);
                  _encryptedDataController.text = encryptedData;
                },
                child: Text('Encrypt'),
              ),
              SizedBox(height: 16),
              TextField(
                controller: _encryptedDataController,
                decoration: InputDecoration(
                  labelText: 'Encrypted Data',
                  border: OutlineInputBorder(),
                ),
              ),
              SizedBox(height: 16),
              ElevatedButton(
                onPressed: () {
                  final encryption = XorEncryption(key: _keyController.text);
                  final decryptedData = encryption.decrypt(_encryptedDataController.text);
                  _decryptedDataController.text = decryptedData;
                },
                child: Text('Decrypt'),
              ),
              SizedBox(height: 16),
              TextField(
                controller: _decryptedDataController,
                decoration: InputDecoration(
                  labelText: 'Decrypted Data',
                  border: OutlineInputBorder(),
                ),
              ),
            ],
          ),
        ),
      ),
    );
  }
}

// The application flow is as follows:
// 1. The user enters a secret key and data to encrypt.
// 2. When the user clicks the "Encrypt" button, the data is encrypted using the XOR encryption algorithm.
// 3. The encrypted data is displayed in the "Encrypted Data" text field.
// 4. When the user clicks the "Decrypt" button, the encrypted data is decrypted using the XOR encryption algorithm.
// 5. The decrypted data is displayed in the "Decrypted Data" text field.

// The XOR encryption algorithm works by performing an exclusive OR operation on each byte of the data with the corresponding byte of the key.
// This produces an encrypted byte that is the same length as the original byte.
// To decrypt the data, the same XOR operation is performed on each byte of the encrypted data with the corresponding byte of the key.
// This produces the original byte, effectively decrypting the data.

// The XOR encryption algorithm is a simple and efficient way to encrypt data, but it is not secure for sensitive data.
// In a real-world application, a more secure encryption algorithm such as AES would be used.

// This demo application is for educational purposes only and should not be used to encrypt sensitive data.
```
