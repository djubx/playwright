import 'package:flutter/material.dart';
import 'package:indonesia/indonesia.dart';

void main() {
  runApp(RealFlutter());
}

class RealFlutter extends StatefulWidget {
  @override
  _RealFlutterState createState() => _RealFlutterState();
}

class _RealFlutterState extends State<RealFlutter> {
  // Create a list to store the provinces
  List<Province> _provinces = [];

  // Create a variable to store the selected province
  Province _selectedProvince;

  // Create a variable to store the selected city
  City _selectedCity;

  // Create a variable to store the selected district
  District _selectedDistrict;

  @override
  void initState() {
    super.initState();

    // Get the list of provinces when the app starts
    _getProvinces();
  }

  // Function to get the list of provinces
  Future<void> _getProvinces() async {
    // Get the list of provinces from the 'indonesia' package
    List<Province> provinces = await Indonesia.getProvinces();

    // Set the list of provinces to the _provinces variable
    setState(() {
      _provinces = provinces;
    });
  }

  // Function to get the list of cities in a province
  Future<void> _getCities(Province province) async {
    // Get the list of cities in the selected province from the 'indonesia' package
    List<City> cities = await Indonesia.getCities(province.id);

    // Show a dialog with the list of cities
    showDialog(
      context: context,
      builder: (BuildContext context) {
        return AlertDialog(
          title: Text('Select City'),
          content: ListView.builder(
            shrinkWrap: true,
            itemCount: cities.length,
            itemBuilder: (BuildContext context, int index) {
              return ListTile(
                title: Text(cities[index].name),
                onTap: () {
                  // Set the selected city to the _selectedCity variable
                  setState(() {
                    _selectedCity = cities[index];
                  });

                  // Close the dialog
                  Navigator.of(context).pop();
                },
              );
            },
          ),
        );
      },
    );
  }

  // Function to get the list of districts in a city
  Future<void> _getDistricts(City city) async {
    // Get the list of districts in the selected city from the 'indonesia' package
    List<District> districts = await Indonesia.getDistricts(city.id);

    // Show a dialog with the list of districts
    showDialog(
      context: context,
      builder: (BuildContext context) {
        return AlertDialog(
          title: Text('Select District'),
          content: ListView.builder(
            shrinkWrap: true,
            itemCount: districts.length,
            itemBuilder: (BuildContext context, int index) {
              return ListTile(
                title: Text(districts[index].name),
                onTap: () {
                  // Set the selected district to the _selectedDistrict variable
                  setState(() {
                    _selectedDistrict = districts[index];
                  });

                  // Close the dialog
                  Navigator.of(context).pop();
                },
              );
            },
          ),
        );
      },
    );
  }

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        appBar: AppBar(
          title: Text('Real Flutter'),
        ),
        body: Center(
          child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: <Widget>[
              // Show the list of provinces
              DropdownButton<Province>(
                value: _selectedProvince,
                onChanged: (Province newValue) {
                  // Set the selected province to the _selectedProvince variable
                  setState(() {
                    _selectedProvince = newValue;
                  });

                  // Get the list of cities in the selected province
                  _getCities(newValue);
                },
                items: _provinces.map((Province province) {
                  return DropdownMenuItem<Province>(
                    value: province,
                    child: Text(province.name),
                  );
                }).toList(),
              ),
              // Show the selected city
              Text(_selectedCity != null ? _selectedCity.name : ''),
              // Show the selected district
              Text(_selectedDistrict != null ? _selectedDistrict.name : ''),
            ],
          ),
        ),
      ),
    );
  }
}

// The app starts by getting the list of provinces from the 'indonesia' package.
// The list of provinces is then shown in a dropdown button.
// When a province is selected, the app gets the list of cities in the selected province.
// The list of cities is then shown in a dialog.
// When a city is selected, the app gets the list of districts in the selected city.
// The list of districts is then shown in a dialog.
// The selected province, city, and district are shown on the screen.
