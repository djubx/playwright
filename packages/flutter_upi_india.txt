import 'package:flutter/material.dart';
import 'package:flutter_upi_india/flutter_upi_india.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'UPI Payment Demo',
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: MyHomePage(),
    );
  }
}

class MyHomePage extends StatefulWidget {
  @override
  _MyHomePageState createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  FlutterUpiIndia _upiIndia = FlutterUpiIndia();

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('UPI Payment Demo'),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: <Widget>[
            Text(
              'UPI Payment Demo',
              style: TextStyle(fontSize: 24),
            ),
            SizedBox(height: 20),
            ElevatedButton(
              onPressed: () async {
                // Step 1: Create a UPI payment request
                UpiPaymentRequest request = UpiPaymentRequest(
                  payeeAddress: 'your-payee-address',
                  payeeName: 'your-payee-name',
                  transactionId: 'your-transaction-id',
                  transactionRefId: 'your-transaction-ref-id',
                  amount: 100,
                );

                // Step 2: Make the UPI payment
                UpiPaymentResponse response = await _upiIndia.makePayment(request);

                // Step 3: Handle the payment response
                if (response.status == UpiPaymentStatus.SUCCESS) {
                  print('Payment successful!');
                } else {
                  print('Payment failed!');
                }
              },
              child: Text('Make UPI Payment'),
            ),
          ],
        ),
      ),
    );
  }
}

// The application flow is as follows:
// 1. The user clicks the "Make UPI Payment" button.
// 2. The UPI payment request is created with the required details.
// 3. The UPI payment is made using the `_upiIndia` instance.
// 4. The payment response is handled based on the status.
// 5. If the payment is successful, a success message is printed.
// 6. If the payment fails, a failure message is printed.

// Final summary: This code demonstrates how to integrate UPI payments into a Flutter application using the `flutter_upi_india` package. The application flow involves creating a UPI payment request, making the payment, and handling the payment response.
