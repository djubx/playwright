import 'package:flutter/material.dart';
import 'package:super_circle/super_circle.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Super Circle Demo',
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: MyHomePage(),
    );
  }
}

class MyHomePage extends StatefulWidget {
  @override
  _MyHomePageState createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Super Circle Demo'),
      ),
      body: Center(
        child: SuperCircle(
          size: 200.0,
          strokeColor: Colors.blue,
          strokeWidth: 10.0,
          segments: [
            SuperCircleSegment(
              color: Colors.red,
              size: 50.0,
              animation: SuperCircleAnimation(
                duration: Duration(milliseconds: 1000),
                curve: Curves.easeInOut,
              ),
            ),
            SuperCircleSegment(
              color: Colors.green,
              size: 75.0,
              animation: SuperCircleAnimation(
                duration: Duration(milliseconds: 1500),
                curve: Curves.easeInOut,
              ),
            ),
          ],
          child: Center(
            child: Text('Super Circle'),
          ),
        ),
      ),
    );
  }
}

// The application flow is as follows:
// 1. The user runs the app and is presented with a Material app with a title and a theme.
// 2. The user is then presented with a Scaffold that contains an AppBar and a body.
// 3. The body of the Scaffold contains a Center widget that contains a SuperCircle widget.
// 4. The SuperCircle widget is configured with a size, stroke color, and stroke width.
// 5. The SuperCircle widget also contains two SuperCircleSegment widgets, each with a different color and size.
// 6. Each SuperCircleSegment widget has an animation that is triggered when the segment is displayed.
// 7. The animations are staggered to create a smooth and engaging user experience.
// 8. The SuperCircle widget also contains a child widget that is displayed in the center of the circle.
// 9. The child widget is a Text widget that displays the text "Super Circle".

// In summary, the code creates a Material app with a Scaffold that contains a SuperCircle widget.
// The SuperCircle widget is configured with two animated segments and a child widget that is displayed in the center of the circle.
// The animations are staggered to create a smooth and engaging user experience.
