import 'package:flutter/material.dart';
import 'package:arabic_roman_conv/arabic_roman_conv.dart';

void main() {
  runApp(RealFlutter());
}

class RealFlutter extends StatefulWidget {
  @override
  _RealFlutterState createState() => _RealFlutterState();
}

class _RealFlutterState extends State<RealFlutter> {
  final _arabicController = TextEditingController();
  final _romanController = TextEditingController();

  void _convertArabicToRoman() {
    final arabic = int.parse(_arabicController.text);
    final roman = arabicToRoman(arabic);
    _romanController.text = roman;
  }

  void _convertRomanToArabic() {
    final roman = _romanController.text;
    final arabic = romanToArabic(roman);
    _arabicController.text = arabic.toString();
  }

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Arabic Roman Converter',
      home: Scaffold(
        appBar: AppBar(
          title: Text('Arabic Roman Converter'),
        ),
        body: Padding(
          padding: const EdgeInsets.all(20.0),
          child: Column(
            children: [
              TextField(
                controller: _arabicController,
                decoration: InputDecoration(
                  labelText: 'Arabic',
                  border: OutlineInputBorder(),
                ),
              ),
              SizedBox(height: 20),
              ElevatedButton(
                onPressed: _convertArabicToRoman,
                child: Text('Convert to Roman'),
              ),
              SizedBox(height: 20),
              TextField(
                controller: _romanController,
                decoration: InputDecoration(
                  labelText: 'Roman',
                  border: OutlineInputBorder(),
                ),
              ),
              SizedBox(height: 20),
              ElevatedButton(
                onPressed: _convertRomanToArabic,
                child: Text('Convert to Arabic'),
              ),
            ],
          ),
        ),
      ),
    );
  }
}

// The app starts by running the main function, which creates a new instance of the RealFlutter widget.
// The RealFlutter widget is a stateful widget that contains two text fields and two buttons.
// The first text field is for entering Arabic numbers, and the second text field is for entering Roman numerals.
// The first button is for converting Arabic numbers to Roman numerals, and the second button is for converting Roman numerals to Arabic numbers.
// When the user clicks the first button, the _convertArabicToRoman function is called, which converts the Arabic number to a Roman numeral using the arabicToRoman function.
// When the user clicks the second button, the _convertRomanToArabic function is called, which converts the Roman numeral to an Arabic number using the romanToArabic function.
// The results of the conversions are displayed in the text fields.
