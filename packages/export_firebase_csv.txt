import 'package:flutter/material.dart';
import 'package:export_firebase_csv/export_firebase_csv.dart';
import 'package:firebase_core/firebase_core.dart';
import 'package:firebase_database/firebase_database.dart';
import 'package:cloud_firestore/cloud_firestore.dart';

void main() async {
  WidgetsFlutterBinding.ensureInitialized();
  await Firebase.initializeApp();
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Export Firebase CSV',
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: RealFlutter(),
    );
  }
}

class RealFlutter extends StatefulWidget {
  @override
  _RealFlutterState createState() => _RealFlutterState();
}

class _RealFlutterState extends State<RealFlutter> {
  final FirebaseDatabase _database = FirebaseDatabase.instance;
  final FirebaseFirestore _firestore = FirebaseFirestore.instance;
  final ExportFirebaseCsv _exportCsv = ExportFirebaseCsv();

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text('Export Firebase CSV'),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: <Widget>[
            ElevatedButton(
              onPressed: () async {
                // Export data from Firebase Realtime Database
                await _exportCsv.export(
                  _database.reference().child('path/to/node'),
                  'output.csv',
                );
              },
              child: Text('Export from Realtime Database'),
            ),
            SizedBox(height: 20),
            ElevatedButton(
              onPressed: () async {
                // Export data from Cloud Firestore
                await _exportCsv.export(
                  _firestore.collection('collectionName'),
                  'output.csv',
                );
              },
              child: Text('Export from Cloud Firestore'),
            ),
          ],
        ),
      ),
    );
  }
}

// Application flow:
// 1. The user clicks on the "Export from Realtime Database" or "Export from Cloud Firestore" button.
// 2. The `_exportCsv.export` method is called with the reference to the node or collection and the output file name.
// 3. The `export` method exports the data from the specified node or collection to a CSV file.
// 4. The CSV file is saved to the device's storage.
// 5. The user can then access the exported data by opening the CSV file.
